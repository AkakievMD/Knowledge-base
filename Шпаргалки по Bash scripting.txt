#!/bin/bash (начало скрипта)  скрипт запускается из папки, где был создан так    ./имя скрипта.sh
--------------------------------------------------------------------------
hello="hello world" (так создаеться текстовая переменная, которая содержит текст с пробелами и без. Чтобы Bash не обрабатывал содержимое, а воспринимал как текст, то нужно ставить одинарные ковычки или ставить знак доллар)
hello=helloworld, 100... (так создаеться переменная, которая не содержит пробелов, а просто значение)
--------------------------------------------------------------------------
if [ условие ]
then
    действия, которые дожны выполниться(команды и т.д.)
else
    альтернативное действие, которое должно выполниться вместо первого действия
fi (пишется в конце конструкции)
Таких конструкций может быть много в скрипте, как отдельных. так и вложенных друг в друга
Так же можно использовать после "if" elif [ условие ] then действе. Это как альтернативное дополнение
--------------------------------------------------------------------------
read (после read пишем переменную, в которую запишем значения во время работы скрипта)(это чтобы в определенный момент иметь возможность ввести какой-нибудь текст с новой строки во время работы скрипта в терминале. В самом скрипте пишется с новой строки. Введенные переменные можно использовать как переменные, со знаком доллар)
break (чтобы прервать работу скрипта)
--------------------------------------------------------------------------
while [ условие ]
do 
   действие, которое будет выполняться, пока условие актуально
done (пишется в конце цикла)
--------------------------------------------------------------------------
myvar=$(( $myvar + любое число )) (это как test = test + text, со значением, к примеру, test = 10 и text = 5, как в языках программирования. Вне цикла создали переменную и в нее внутри цикла записываем новое значение, пока актуально условие)
--------------------------------------------------------------------------
select переменная in список значений (Иметирует меню выбора. Список значений через пробел, который попадет в переменную можно будет выбрать)
do команды
done
--------------------------------------------------------------------------
case $переменная in     (Это читатается так... Если в переменной есть одно и выбранных значений, то выполнить команду в выбранном значении)
  значение1) команды ;;
  значение2) команды ;;
  *) команды, если значения нет в списке ;;
esac
---------------------------------------------------------------------------
select number in one two three stop   (пример работы меню с выбором значения и выполнения последующих команд в этих значениях)
do case $number in
     one) echo 1 ;;
     two) echo 2 ;;
     three) echo 3 ;;
     stop) break ;;
     *) echo Wrong option ;;
    esac
done
----------------------------------------------------------------------------
for переменная in $(здесь пишем команду и значение или ряд значений, или файл...);
do команда
done
----------------------------------------------------------------------------
