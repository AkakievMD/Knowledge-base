---
- name: install 
  hosts: hostname
  become: yes 

  tasks:
  - yum: 
     name: https://repo.zabbix.com/zabbix/5.0/rhel/7/x86_64/zabbix-release-5.0-1.el7.noarch.rpm 
     state: present

  - shell: yum clean all

  - yum:
     name: "{{item}}"
     state: latest
    loop: 
      - centos-release-scl
      - zabbix-agent
      - iptables-services  
  
  - systemd:
      name: iptables
      state: started
      enabled: yes 

  - systemd:
      name: firewalld
      state: stopped
      enabled: no

  - systemd:
      name: zabbix-agent
      state: started  
      enabled: yes 
  
  - lineinfile:
      path: /etc/selinux/config
      regexp: '^SELINUX=enforcing'
      line: SELINUX=disabled
      owner: root
      group: root
      firstmatch: yes
  
  - lineinfile:
      path: /etc/sysconfig/iptables
      insertafter: '^-A INPUT -p tcp -m state --state NEW -m tcp --dport 22 -j ACCEPT'
      line: "{{item}}"
      owner: root
      group: root
      firstmatch: yes 
    loop:
      - -A INPUT -p tcp -m tcp --dport 10050 -j ACCEPT
      - -A INPUT -p tcp -m tcp --dport 10051 -j ACCEPT
      - -A OUTPUT -p tcp -m tcp --dport 10050 -j ACCEPT
      - -A OUTPUT -p tcp -m tcp --dport 10051 -j ACCEPT

  - name:  
    service: 
      name: iptables 
      state: restarted 

  - lineinfile:
      path: /etc/zabbix/zabbix_agentd.conf
      regexp: '^Server=127\.0\.0\.1'
      line: Server=ip-address zabbix-server
      owner: root
      group: root
      firstmatch: yes

  - lineinfile:
      path: /etc/zabbix/zabbix_agentd.conf
      regexp: '^# HostMetadata='
      line: 'HostMetadata=linux'  (это просто метка по которой опция авто добавления в zabbix будет искать хост)
      owner: root
      group: root
      firstmatch: yes

  - lineinfile:
      path: /etc/zabbix/zabbix_agentd.conf
      regexp: '^ServerActive=127\.0\.0\.1'
      line: ServerActive=ip-address zabbix-server (если не используете такой тип проверок то закоментируйте это поле)
      owner: root
      group: root
      firstmatch: yes

  - lineinfile:
      path: /etc/zabbix/zabbix_agentd.conf
      regexp: '^Hostname=Zabbix server'
      line: Hostname=имя хоста
      owner: root
      group: root
      firstmatch: yes
  
  - name: start web
    service:
      name: zabbix-agent
      state: restarted

  - shell: reboot
