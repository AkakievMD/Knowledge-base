---
- name: install
  hosts: zabbix
  become: yes

  tasks:
  - name: Install Zabbix
    dnf:
      name: 'https://repo.zabbix.com/zabbix/5.0/rhel/8/x86_64/zabbix-release-5.0-1.el8.noarch.rpm'
      disable_gpg_check: yes
      state: present

  - shell: yum clean all

  - lineinfile:
      path: /etc/yum.repos.d/zabbix.repo
      regexp: '^enabled=0'
      line: enabled=1
      firstmatch: yes

  - yum:
     name: "{{item}}"
     state: latest
    loop:
      - python3-pip
      - zabbix-server-mysql
      - zabbix-agent
      - mariadb-server
      - zabbix-web-mysql
      - zabbix-apache-conf

  - name: Install PyMySQL
    pip:
      name: PyMySQL

  - systemd:
      name: mariadb
      state: started
      enabled: yes

  - lineinfile:
      path: /etc/selinux/config
      regexp: '^SELINUX=enforcing'
      line: SELINUX=disabled
      firstmatch: yes

  - name: Creating database
    mysql_db:
     name: zabbix_db
     encoding: utf8
     collation: utf8_bin
     state: present

  - name: Create database user
    mysql_user:
     name: zabbix_user
     password: passw0rd123
     priv: '*.*:ALL,GRANT'
     state: present

  - shell: zcat /usr/share/doc/zabbix-server-mysql*/create.sql.gz | mysql -u zabbix_user -p zabbix_db -ppassw0rd123
    args:
      executable: /bin/bash

  - lineinfile:
      path: /etc/zabbix/zabbix_server.conf
      regexp: '^DBName=zabbix'
      line: DBName=zabbix_db
      firstmatch: yes

  - lineinfile:
      path: /etc/zabbix/zabbix_server.conf
      regexp: '^DBUser=zabbix'
      line: DBUser=zabbix_user
      firstmatch: yes

  - lineinfile:
      path: /etc/zabbix/zabbix_server.conf
      insertafter: '^# DBPassword='
      line: DBPassword=passw0rd123
      firstmatch: yes

  - lineinfile:
      path: /etc/php-fpm.d/zabbix.conf
      regexp: '^; php_value\[date.timezone\] = Europe/Riga'
      line: 'php_value[date.timezone] = Europe/Moscow'
      firstmatch: yes

  - firewalld:
      port: 10050-10051/tcp
      permanent: yes
      immediate: yes
      state: enabled

  - firewalld:
      service: "{{item}}"
      permanent: yes
      immediate: yes
      state: enabled
    loop:
     - https
     - http

  - name:
    systemd:
      name: "{{item}}"
      enabled: yes
    loop:
     - zabbix-server
     - zabbix-agent
     - httpd
     - php-fpm

  - shell: reboot
